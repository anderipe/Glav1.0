361:     /**
362:      * Adjust the brightness of a color
363:      *
364:      * @param   string      $hex    The colour as an RGB value (e.g. FF00CCCC or CCDDEE
365:      * @param   float       $adjustPercentage   The percentage by which to adjust the colour as a float from -1 to 1
366:      * @return  string      The adjusted colour as an RGB value (e.g. FF00CCCC or CCDDEE
367:      */
368:     public static function changeBrightness($hex, $adjustPercentage) {
369:         $red    = self::getRed($hex,false);
370:         $green  = self::getGreen($hex,false);
371:         $blue   = self::getBlue($hex,false);
372:         if ($adjustPercentage > 0) {
373:             $red    += (255 - $red) * $adjustPercentage;
374:             $green  += (255 - $green) * $adjustPercentage;
375:             $blue   += (255 - $blue) * $adjustPercentage;
376:         } else {
377:             $red    += $red * $adjustPercentage;
378:             $green  += $green * $adjustPercentage;
379:             $blue   += $blue * $adjustPercentage;
380:         }
381: 
382:         if ($red < 0) $red = 0;
383:         elseif ($red > 255) $red = 255;
384:         if ($green < 0) $green = 0;
385:         elseif ($green > 255) $green = 255;
386:         if ($blue < 0) $blue = 0;
387:         elseif ($blue > 255) $blue = 255;
388: 
389:         return strtoupper(  str_pad(dechex($red), 2, '0', 0) .
390:                             str_pad(dechex($green), 2, '0', 0) .
391:                             str_pad(dechex($blue), 2, '0', 0)
