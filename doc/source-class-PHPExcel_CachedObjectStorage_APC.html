 38:     /**
 39:      * Prefix used to uniquely identify cache data for this worksheet
 40:      *
 41:      * @access  private
 42:      * @var string
 43:      */
 44:     private $_cachePrefix = null;
 45: 
 46:     /**
 47:      * Cache timeout
 48:      *
 49:      * @access  private
 50:      * @var integer
 51:      */
 52:     private $_cacheTime = 600;
 53: 
 54: 
 55:     /**
 56:      * Store cell data in cache for the current cell object if it's "dirty",
 57:      *     and the 'nullify' the current cell object
 58:      *
 59:      * @access  private
 60:      * @return  void
 61:      * @throws  Exception
 62:      */
 63:     private function _storeData() {
 64:         if ($this->_currentCellIsDirty) {
 65:             $this->_currentObject->detach();
 66: 
 67:             if (!apc_store($this->_cachePrefix.$this->_currentObjectID.'.cache',serialize($this->_currentObject),$this->_cacheTime)) {
 68:                 $this->__destruct();
 69:                 throw new Exception('Failed to store cell '.$this->_currentObjectID.' in APC');
 70:             }
 71:             $this->_currentCellIsDirty = false;
 72:         }
 73:         $this->_currentObjectID = $this->_currentObject = null;
 74:     }   //  function _storeData()
 75: 
 76: 
 77:     /**
 78:      * Add or Update a cell in cache identified by coordinate address
 79:      *
 80:      * @access  public
 81:      * @param   string          $pCoord     Coordinate address of the cell to update
 82:      * @param   PHPExcel_Cell   $cell       Cell to update
 83:      * @return  void
 84:      * @throws  Exception
 85:      */
 86:     public function addCacheData($pCoord, PHPExcel_Cell $cell) {
 87:         if (($pCoord !== $this->_currentObjectID) && ($this->_currentObjectID !== null)) {
 88:             $this->_storeData();
 89:         }
 90:         $this->_cellCache[$pCoord] = true;
 91: 
 92:         $this->_currentObjectID = $pCoord;
 93:         $this->_currentObject = $cell;
 94:         $this->_currentCellIsDirty = true;
 95: 
 96:         return $cell;
 97:     }   //  function addCacheData()
 98: 
 99: 
100:     /**
101:      * Is a value set in the current PHPExcel_CachedObjectStorage_ICache for an indexed cell?
102:      *
103:      * @access  public
104:      * @param   string      $pCoord     Coordinate address of the cell to check
105:      * @return  void
106:      * @return  boolean
107:      */
108:     public function isDataSet($pCoord) {
109:         //  Check if the requested entry is the current object, or exists in the cache
110:         if (parent::isDataSet($pCoord)) {
111:             if ($this->_currentObjectID == $pCoord) {
112:                 return true;
113:             }
114:             //  Check if the requested entry still exists in apc
115:             $success = apc_fetch($this->_cachePrefix.$pCoord.'.cache');
116:             if ($success === false) {
117:                 //  Entry no longer exists in APC, so clear it from the cache array
118:                 parent::deleteCacheData($pCoord);
119:                 throw new Exception('Cell entry '.$pCoord.' no longer exists in APC');
120:             }
121:             return true;
122:         }
123:         return false;
124:     }   //  function isDataSet()
125: 
126: 
127:     /**
128:      * Get cell at a specific coordinate
129:      *
130:      * @access  public
131:      * @param   string          $pCoord     Coordinate of the cell
132:      * @throws  Exception
133:      * @return  PHPExcel_Cell   Cell that was found, or null if not found
134:      */
135:     public function getCacheData($pCoord) {
136:         if ($pCoord === $this->_currentObjectID) {
137:             return $this->_currentObject;
138:         }
139:         $this->_storeData();
140: 
141:         //  Check if the entry that has been requested actually exists
142:         if (parent::isDataSet($pCoord)) {
143:             $obj = apc_fetch($this->_cachePrefix.$pCoord.'.cache');
144:             if ($obj === false) {
145:                 //  Entry no longer exists in APC, so clear it from the cache array
146:                 parent::deleteCacheData($pCoord);
147:                 throw new Exception('Cell entry '.$pCoord.' no longer exists in APC');
148:             }
149:         } else {
150:             //  Return null if requested entry doesn't exist in cache
151:             return null;
152:         }
153: 
154:         //  Set current entry to the requested entry
155:         $this->_currentObjectID = $pCoord;
156:         $this->_currentObject = unserialize($obj);
157:         //  Re-attach the parent worksheet
158:         $this->_currentObject->attach($this->_parent);
159: 
160:         //  Return requested entry
161:         return $this->_currentObject;
162:     }   //  function getCacheData()
163: 
164: 
165:     /**
166:      * Delete a cell in cache identified by coordinate address
167:      *
168:      * @access  public
169:      * @param   string          $pCoord     Coordinate address of the cell to delete
170:      * @throws  Exception
171:      */
172:     public function deleteCacheData($pCoord) {
173:         //  Delete the entry from APC
174:         apc_delete($this->_cachePrefix.$pCoord.'.cache');
175: 
176:         //  Delete the entry from our cell address array
177:         parent::deleteCacheData($pCoord);
178:     }   //  function deleteCacheData()
179: 
180: 
181:     /**
182:      * Clone the cell collection
183:      *
184:      * @access  public
185:      * @param   PHPExcel_Worksheet  $parent     The new worksheet
186:      * @return  void
187:      */
188:     public function copyCellCollection(PHPExcel_Worksheet $parent) {
189:         parent::copyCellCollection($parent);
190:         //  Get a new id for the new file name
191:         $baseUnique = $this->_getUniqueID();
192:         $newCachePrefix = substr(md5($baseUnique),0,8).'.';
193:         $cacheList = $this->getCellList();
194:         foreach($cacheList as $cellID) {
195:             if ($cellID != $this->_currentObjectID) {
196:                 $obj = apc_fetch($this->_cachePrefix.$cellID.'.cache');
197:                 if ($obj === false) {
198:                     //  Entry no longer exists in APC, so clear it from the cache array
199:                     parent::deleteCacheData($cellID);
200:                     throw new Exception('Cell entry '.$cellID.' no longer exists in APC');
201:                 }
202:                 if (!apc_store($newCachePrefix.$cellID.'.cache',$obj,$this->_cacheTime)) {
203:                     $this->__destruct();
204:                     throw new Exception('Failed to store cell '.$cellID.' in APC');
205:                 }
206:             }
207:         }
208:         $this->_cachePrefix = $newCachePrefix;
209:     }   //  function copyCellCollection()

212:     /**
213:      * Clear the cell collection and disconnect from our parent
214:      *
215:      * @access  public
216:      * @return  void
217:      */
218:     public function unsetWorksheetCells() {
219:         if ($this->_currentObject !== NULL) {
220:             $this->_currentObject->detach();
221:             $this->_currentObject = $this->_currentObjectID = null;
222:         }
223: 
224:         //  Flush the APC cache
225:         $this->__destruct();
226: 
227:         $this->_cellCache = array();
228: 
229:         //  detach ourself from the worksheet, so that it can then delete this object successfully
230:         $this->_parent = null;
231:     }   //  function unsetWorksheetCells()
232:

234:     /**
235:      * Initialise this new cell collection
236:      *
237:      * @access  public
238:      * @param   PHPExcel_Worksheet  $parent     The worksheet for this cell collection
239:      * @param   array of mixed      $arguments  Additional initialisation arguments
240:      */
241:     public function __construct(PHPExcel_Worksheet $parent, $arguments) {
242:         $cacheTime  = (isset($arguments['cacheTime']))  ? $arguments['cacheTime']   : 600;
243: 
244:         if ($this->_cachePrefix === NULL) {
245:             $baseUnique = $this->_getUniqueID();
246:             $this->_cachePrefix = substr(md5($baseUnique),0,8).'.';
247:             $this->_cacheTime = $cacheTime;
248: 
249:             parent::__construct($parent);
250:         }
251:     }   //  function __construct()

254:     /**
255:      * Destroy this cell collection
256:      *
257:      * @access  public
258:      */
259:     public function __destruct() {
260:         $cacheList = $this->getCellList();
261:         foreach($cacheList as $cellID) {
262:             apc_delete($this->_cachePrefix.$cellID.'.cache');
263:         }
264:     }   //  function __destruct()
