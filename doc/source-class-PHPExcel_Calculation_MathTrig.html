  51:     private static function _factors($value) {
  52:         $startVal = floor(sqrt($value));
  53: 
  54:         $factorArray = array();
  55:         for ($i = $startVal; $i > 1; --$i) {
  56:             if (($value % $i) == 0) {
  57:                 $factorArray = array_merge($factorArray,self::_factors($value / $i));
  58:                 $factorArray = array_merge($factorArray,self::_factors($i));
  59:                 if ($i <= sqrt($value)) {
  60:                     break;
  61:                 }
  62:             }
  63:         }
  64:         if (!empty($factorArray)) {
  65:             rsort($factorArray);
  66:             return $factorArray;
  67:         } else {
  68:             return array((integer) $value);
  69:         }
  70:     }   //  function _factors()

 284:     /**
 285:      * GCD
 286:      *
 287:      * Returns the greatest common divisor of a series of numbers
 288:      *
 289:      * @param   $array  Values to calculate the Greatest Common Divisor
 290:      * @return  int     Greatest Common Divisor
 291:      */
 292:     public static function GCD() {
 293:         $returnValue = 1;
 294:         $allPoweredFactors = array();
 295:         // Loop through arguments
 296:         foreach(PHPExcel_Calculation_Functions::flattenArray(func_get_args()) as $value) {
 297:             if (!is_numeric($value)) {
 298:                 return PHPExcel_Calculation_Functions::VALUE();
 299:             } elseif ($value == 0) {
 300:                 break;
 301:             } elseif($value < 0) {
 302:                 return PHPExcel_Calculation_Functions::NaN();
 303:             }
 304:             $myFactors = self::_factors($value);
 305:             $myCountedFactors = array_count_values($myFactors);
 306:             $allValuesFactors[] = $myCountedFactors;
 307:         }
 308:         $allValuesCount = count($allValuesFactors);
 309:         $mergedArray = $allValuesFactors[0];
 310:         for ($i=1;$i < $allValuesCount; ++$i) {
 311:             $mergedArray = array_intersect_key($mergedArray,$allValuesFactors[$i]);
 312:         }
 313:         $mergedArrayValues = count($mergedArray);
 314:         if ($mergedArrayValues == 0) {
 315:             return $returnValue;
 316:         } elseif ($mergedArrayValues > 1) {
 317:             foreach($mergedArray as $mergedKey => $mergedValue) {
 318:                 foreach($allValuesFactors as $highestPowerTest) {
 319:                     foreach($highestPowerTest as $testKey => $testValue) {
 320:                         if (($testKey == $mergedKey) && ($testValue < $mergedValue)) {
 321:                             $mergedArray[$mergedKey] = $testValue;
 322:                             $mergedValue = $testValue;
 323:                         }
 324:                     }
 325:                 }
 326:             }
 327: 
 328:             $returnValue = 1;
 329:             foreach($mergedArray as $key => $value) {
 330:                 $returnValue *= pow($key,$value);
 331:             }
 332:             return $returnValue;
 333:         } else {
 334:             $keys = array_keys($mergedArray);
 335:             $key = $keys[0];
 336:             $value = $mergedArray[$key];
 337:             foreach($allValuesFactors as $testValue) {
 338:                 foreach($testValue as $mergedKey => $mergedValue) {
 339:                     if (($mergedKey == $key) && ($mergedValue < $value)) {
 340:                         $value = $mergedValue;
 341:                     }
 342:                 }
 343:             }
 344:             return pow($key,$value);
 345:         }
 346:     }   //  function GCD()
